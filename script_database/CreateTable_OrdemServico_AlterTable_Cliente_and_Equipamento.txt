
create table ordemServico(
id serial,
clienteId INT,
equipamentoId INT,
servico VARCHAR(100),
criticidade VARCHAR(50),
datainicio TIMESTAMP,
datafim TIMESTAMP,
primary key (id),
constraint fk_cliente FOREIGN KEY(clienteId) references cliente(id),
constraint fk_equipamento FOREIGN KEY(equipamentoId) references equipamento(id)
)
